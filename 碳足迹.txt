<script>
  function switchNordicView(viewId) {
    // 移除所有視圖的 active 狀態
    document.querySelectorAll('.nordic-eco-grid').forEach(function(grid) {
      grid.classList.remove('active-view');
    });
    // 移除所有按鈕的 active 狀態
    document.querySelectorAll('.nordic-tab-btn').forEach(function(btn) {
      btn.classList.remove('active');
    });
    
    // 激活當前選擇的視圖與按鈕
    document.getElementById(viewId).classList.add('active-view');
    event.currentTarget.classList.add('active');
  }
</script>
